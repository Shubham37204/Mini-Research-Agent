APP_NAME=mini-research-agent  → project name, used in logs
DEBUG=True                    → show detailed errors, auto-reload on code change
                                set False in production
HOST=0.0.0.0                  → listen on ALL network interfaces
                                (not just localhost, so other devices can hit it)
PORT=8000                     → FastAPI runs here → http://localhost:8000
LLM_MODEL=llama-3.3-70b-versatile  → which Groq model to use
                                      70b = 70 billion params, smart + fast
LLM_TEMPERATURE=0.7                → creativity level
                                      0.0 = robotic, exact
                                      1.0 = creative, random
                                      0.7 = balanced (good for research agent)
CHROMA_PERSIST_DIR=./chroma_db   → folder where ChromaDB saves vectors on disk
                                   persists between restarts (not in-memory)

EMBEDDING_MODEL=all-MiniLM-L6-v2 → converts text → numbers (vectors)
                                   small, fast, runs LOCAL, no API needed

CHUNK_SIZE=500                    → split docs into 500-char pieces before storing
                                   smaller = more precise retrieval

CHUNK_OVERLAP=50                  → each chunk shares 50 chars with next chunk
                                   prevents losing context at chunk boundaries
